                          示例 #

                          运行此代码

                          #include <iostream>
int main()
{
    std::cout << "Printable ASCII:" << std::endl;
    for (char i = 32; i < 127; ++i) {
        std::cout << i << ' ';
        if (i % 16 == 15)
            std::cout << std::endl;
    }
}
